 47      """Hashes a file using OpenSubtitle's method. 
 48   
 49      > In natural language it calculates: size + 64bit chksum of the first and 
 50      > last 64k (even if they overlap because the file is smaller than 128k). 
 51   
 52      A slightly more Pythonic version of the Python solution on.. 
 53      http://trac.opensubtitles.org/projects/opensubtitles/wiki/HashSourceCodes 
 54      """ 
 55      longlongformat = 'q' 
 56      bytesize = struct.calcsize(longlongformat) 
 57   
 58      f = open(name, "rb") 
 59   
 60      filesize = os.path.getsize(name) 
 61      fhash = filesize 
 62   
 63      if filesize < 65536 * 2: 
 64         raise ValueError("File size must be larger than %s bytes (is %s)" % (65536 * 2, filesize)) 
 65   
 66      for x in range(65536 / bytesize): 
 67          buf = f.read(bytesize) 
 68          (l_value,) = struct.unpack(longlongformat, buf) 
 69          fhash += l_value 
 70          fhash = fhash & 0xFFFFFFFFFFFFFFFF # to remain as 64bit number 
 71   
 72   
 73      f.seek(max(0, filesize - 65536), 0) 
 74      for x in range(65536 / bytesize): 
 75          buf = f.read(bytesize) 
 76          (l_value,) = struct.unpack(longlongformat, buf) 
 77          fhash += l_value 
 78          fhash = fhash & 0xFFFFFFFFFFFFFFFF 
 79   
 80      f.close()
